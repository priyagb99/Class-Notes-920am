import datetime
import numpy as np
import pandas as pd
import pandas_datareader as pdr  

start = datetime.datetime(1960, 1, 1)
end = datetime.datetime(2019, 1, 27)
realGDP_df = pdr.data.DataReader(['GDPC1'], 'fred', start, end)
realGDP_df['Year'] = realGDP_df.index.year

##create annual real GDP dataframe
annual = realGDP_df.groupby(["Year"])["GDPC1"].sum()
annual_df=pd.DataFrame(annual.reset_index(name = "Annual_GDP"))

## find percentage change
annual_df=annual_df.assign(Pct_Change=annual_df['Annual_GDP'].pct_change())
annual_df
